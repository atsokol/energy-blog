```js
{
  // ---- Prep with year & month
  const data = volumes
    .filter(d => d.unit === "тис. тонн")
    .map(d => ({
      ...d,
      port_name: otherPortsMap.get(d.port_name),
      cargo_type: otherCargoesMap.get(d.cargo_type),
      year: d.date.getUTCFullYear(),
      month: d.date.getUTCMonth() + 1
    }))
    .filter(d => d.year >= 2019 && d.year !== 2022);

  // ---- Latest observation (defines "current" year/month)
  const maxDate = d3.max(data, d => d.date);
  const maxYear = maxDate.getUTCFullYear();
  const maxMonth = maxDate.getUTCMonth() + 1;

  // ---- Decide whether to show YoY YTD comparison
  // Show only if latest year is incomplete (1–11 months) AND both years have data in the latest month.
  const monthsThisYear = new Set(data.filter(d => d.year === maxYear).map(d => d.month));
  const hasIncompleteYear = maxMonth < 12 && monthsThisYear.size < 12;

  const hasCurrentMonthThisYear = data.some(d => d.year === maxYear && d.month === maxMonth);
  const hasCurrentMonthPrevYear = data.some(d => d.year === (maxYear - 1) && d.month === maxMonth);

  const showYTD = hasIncompleteYear && hasCurrentMonthThisYear && hasCurrentMonthPrevYear;

  // ---- Full-year panel
  // If YTD is shown: exclude incomplete current year from FY.
  // If YTD is NOT shown: include maxYear (it should be complete).
  const fyMaxYearExclusive = showYTD ? maxYear : (maxYear + 1);

  const fullYears = d3.flatRollup(
      data.filter(d => d.year < fyMaxYearExclusive),
      v => d3.sum(v, d => d.volume),
      d => d.cargo_type, d => String(d.year)
    )
    .map(([cat, x, volume]) => ({ panel: "FY", cat, x: String(x), volume }));

  const fullTotals = d3.rollups(
      fullYears,
      v => d3.sum(v, d => d.volume),
      d => d.x
    ).map(([x, volume]) => ({ panel: "FY", x: String(x), volume }));

  // ---- YTD panel (only if showYTD)
  const ytd = (yr) => d3.flatRollup(
      data.filter(d => d.year === yr && d.month <= maxMonth),
      v => d3.sum(v, d => d.volume),
      d => d.cargo_type
    ).map(([cat, volume]) => ({ panel: "YTD", cat, x: `${maxMonth}m${yr}`, volume }));

  const ytdBars = showYTD ? [...ytd(maxYear - 1), ...ytd(maxYear)] : [];

  const ytdTotals = showYTD
    ? d3.rollups(
        ytdBars,
        v => d3.sum(v, d => d.volume),
        d => d.x
      ).map(([x, volume]) => ({ panel: "YTD", x, volume }))
    : [];

  // ---- Combined datasets
  const dataByCatPeriod = [...fullYears, ...ytdBars];
  const dataByPeriod = [...fullTotals, ...ytdTotals];

  // ---- Color legend order (by total across what’s displayed)
  const colorLegendOrder = d3.groupSort(
    dataByCatPeriod,
    v => -d3.sum(v, d => d.volume),
    d => d.cat
  );

  // ---- X domain
  const width = 650;
  const marginLeft = 50;
  const padding = 0.1;
  const align = 0.5;

  const fyYears = d3.range(2019, fyMaxYearExclusive).map(d => String(d)); // includes maxYear only if complete
  const ytdX = showYTD ? [`${maxMonth}m${maxYear - 1}`, `${maxMonth}m${maxYear}`] : [];
  const xDomain = [...fyYears, ...ytdX];

  // ---- Optional rectangle behind YTD
  const xScale = d3.scaleBand()
    .domain(xDomain)
    .range([marginLeft, width])
    .padding(padding)
    .align(align);

  const maxVol = d3.max(dataByPeriod, d => d.volume) ?? 0;

  const ytdRectMark = showYTD
    ? (() => {
        const left  = xScale(ytdX[0]) - xScale.bandwidth() * 1.05;
        const right = xScale(ytdX[1]) + xScale.bandwidth() * 1.05;
        return Plot.rect([{}], {
          y1: -maxVol * 0.03,
          y2: maxVol * 1.03,
          frameAnchor: "top-left",
          insetLeft: left,
          insetRight: width - right,
          fill: "#FDDDE6",
          fillOpacity: 0.6,
          r: 10
        });
      })()
    : null;

  return Plot.plot({
    width,
    marginLeft,
    x: { type: "band", domain: xDomain, label: null, padding, align },
    y: { grid: true, label: `'000 tonnes'` },
    color: { legend: true, domain: colorLegendOrder, range: d3.schemeTableau10 },
    marks: [
      ...(ytdRectMark ? [ytdRectMark] : []),

      Plot.barY(dataByCatPeriod, {
        x: "x",
        y: "volume",
        fill: "cat",
        order: "value",
        reverse: true,
        fillOpacity: 0.8,
        tip: {
          format: {
            fill: d => d,
            volume: d => d3.format(",.0f")(d)
          }
        }
      }),

      Plot.text(dataByPeriod, {
        x: "x",
        y: "volume",
        text: d => d3.format(",.0f")(d.volume),
        dy: -10
      })
    ]
  });
}
```

```js
treemapChart = {

    .filter(d => d.unit === "тис. тонн")
    .map(d => ({
      ...d,
      port_name: otherPortsMap.get(d.port_name),
      year: d.date.getUTCFullYear()
    }))
    .filter(d => selectCargo === "All" ? true : d.cargo_type === selectCargo)
    .filter(d => d.year === selectYear)      // filter by year
    .filter(d => d.year >= 2019 && d.year != 2022) 
    .map(d => ({...d, name: `${d.port_name}~${d.port_operator}`}))

  const data_treemap = d3.rollups(data_filt, 
                             v => d3.sum(v, d => d.volume), 
                             d => d.name)
    .map(([name, amount]) => ({name, amount}))

  const total = Math.max(1, d3.sum(data_treemap, d => d.amount)); // guard against 0
  
  return Treemap(data_treemap, {
    path: (d) => d.name.replaceAll("~", "/"), // 
    label: (d, n) => (n.value / total) >= 0.01 ? 
      [...d.name.split("~").pop().split(/(?=[A-Z][a-z])/g),  d3.format(",.0f")(n.value)].join("\n") : 
      "",
    group: (d) => d.name.split("~")[0], // for color
    value: (d) => d?.amount, // area of each rect
    title: (d, n) => [n.id, n.value.toLocaleString("en")].join("\n"), // hover text
    width: 800,
    height: 300,
    colors: d3.schemeTableau10,
    fillOpacity: 0.7
  })
```

```js
  );
  
  // 2) Assign each row to an x-bin start (x1), then roll up volume per (year, cargo, bin)
  const binned = d3.flatRollup(
    raw.map(d => {
      const x1 = Math.floor(d.dwt / interval) * interval;
      return {...d, x1};
    }),
    v => d3.sum(v, d => d.volume), // tonnage in this bin
    d => d.year,
    d => d.cargo_type,
    d => d.x1
  ).map(([year, cargo, x1, volSum]) => {
    const total = totals.get(year)?.get(cargo) ?? 0;
    return {
      y: year,
      fy: cargo,
      x1,
      x2: x1 + interval,
      vol: volSum,
      share: total ? volSum / total : 0,
    };
  });

  // 2) Add cumulative volume & cumulative share within each (year, cargo)
const binnedCum = d3.flatGroup(binned, d => d.y, d => d.fy).flatMap(([year, cargo, rows]) => {
    rows.sort((a, b) => d3.ascending(a.x1, b.x1));        // ensure left→right order
    let running = 0;
    let previous
    const total = totals.get(year)?.get(cargo) ?? 0;
    return rows.map(r => {
      previous = running;
      running += r.vol;
      return {
        ...r,
        cshare: total ? running / total : 0,               // cumulative share (0..1)
        cshareLag: total ? previous / total : 0
      };
    });
  });
  
  // 3) Plot the heatmap of bin share of total tonnage
  return Plot.plot({
    title: "Distribution of tonnage by ship size",
    width: 550,
    marginLeft: 50,
    marginRight: 150,
    x: {label: "Deadweight ('000 tonnes)", grid: true},
    y: {label: null, tickFormat: String},
    fy: {label: null, domain: topNCargoes},
    color: {scheme: "YlOrRd", domain: [0, 1], label: "Share of total tonnage", legend: true, tickFormat: d3.format(".0%")},
    marks: [
      Plot.rect(binned, {
        x1: "x1",
        x2: "x2",
        y: "y",
        fy: "fy",
        fill: "share",
        tip: true,
        inset: 0.5
      }),
      Plot.dot(binnedCum, {
        x: d => (d.x1 + d.x2) / 2,
        y: "y",
        fy: "fy",
        fill: "steelblue",
        fillOpacity: d => (d.cshare >= 0.5) && (d.cshareLag <= 0.5) ? 1 : 0
      }
        
      )
    ]
  });

}
```
